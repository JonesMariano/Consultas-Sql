SELECT A.CLI_ST_NOME CLIENTE,
       A.PRO_ST_NOME PRODUTO,
       A.PVE_RE_QUANTIDADE QUANTIDADE, 
       A.VEN_RE_VALOR VALOR
FROM 
( SELECT VEN.VEN_IN_SEQUENCIAL,
         VEN.CLI_IN_SEQUENCIAL,
         CLI.CLI_ST_NOME,
         PRO.PRO_ST_NOME,
         PVE.PVE_RE_QUANTIDADE, 
         VEN.VEN_RE_VALOR
FROM TST_VENDA VEN
INNER JOIN TST_CLIENTE CLI ON CLI.CLI_IN_SEQUENCIAL = VEN.CLI_IN_SEQUENCIAL
INNER JOIN TST_PRODUTOVENDIDO PVE ON PVE.VEN_IN_SEQUENCIAL = VEN.VEN_IN_SEQUENCIAL
INNER JOIN TST_PRODUTO PRO ON PVE.PRO_IN_SEQUENCIAL = PRO.PRO_IN_SEQUENCIAL
)A
ORDER BY A.CLI_ST_NOME ASC, A.PVE_RE_QUANTIDADE DESC
